//推荐使用下面这个方法,图片的名字是自己定义,使用起来比较方便
function loading(loadingImgs,succFn){
//定义一个空的对象
var imgs = {};

var count = 0;
var allNum = 0;
//每循环一次allNum++,获得总的图片数
for (var i in loadingImgs){
allNum++;
}
//将每一张图片进行load
for (var i in loadingImgs){
//将数组里的每个都变成图片对象
imgs[i] = new Image();
imgs[i].src = loadingImgs[i];
imgs[i].onload = function (){

count++;
if (count>=allNum){
//加载完之后运行函数,接受imgs数组
succFn(imgs);
}
}
}
}
var loadingImgs = {
//这里输入要加载的图片链接及名字
background:”img/background.png”,
bomb:”img/bomb.png”,
bullet:”img/bullet1.png”
}

loading(loadingImgs,function (imgs){
//回调函数会返回预加载好的imgs数组
//加载完之后就可以直接使用了,imgs.background就是图片对象,可以获取它的宽高等属性
console.log(imgs.background.width);

});
